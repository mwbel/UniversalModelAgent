from realtime_calculator import RealtimePlanetaryCalculator
from datetime import datetime

# 初始化
calc = RealtimePlanetaryCalculator('de440.bsp')

# 计算地球位置
pos, vel, dist = calc.compute_position('earth', datetime.utcnow())
print(f"地球距离太阳: {dist:.6f} AU")

# 批量计算
planets = ['mercury', 'venus', 'earth', 'mars']
results = calc.compute_multiple(planets)
